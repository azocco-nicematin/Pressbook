/**
 * Controller pour les clôtures, utilisé pour les saisies, rapport et tableaux de bord
 * @module clotureController
 */
const Cloture = require('../models/cloture');
const Titre = require('../models/titre');
const moment = require('../moment-with-locales');

/** Création d'un objet cloture lors d'une fin de service
 * @param {object} infosConnectedUser - informations Google de l'user utilisées pour définir l'attribut login
 * @param {object} service - service de l'utilisateur contenu dans le cookie de session
 */
const createCloture = async (req, res, infosConnectedUser, service) => {
    try {
        const serviceJournee = new Cloture({
            notation: req.body.notation,
            statut: "clos",
            heure_cloture: req.body.heure_cloture,
            service: infosConnectedUser.service,
            serviceUser : service,
            date: moment(new Date()).add(2, 'h'),
            login: infosConnectedUser.adresseMail.replace("@nicematin.fr", "")
        });
        await serviceJournee.save();
        return serviceJournee;
    } catch (err) {
        console.log(err);
        res.status(400).json({
            status: 'fail',
            message: err
        });
    }
};

/** Récupération des clôtures contenues dans une période renseignée
 * @param {object} req - contient les dates de début et de fin
 * @param {object} res - objet renvoyé contenant les données demandées
 */

const getCloture = async (req, res) => {
    try {
        let dateDebut = moment(req.body.dateDebut);
        let dateFin = (moment(req.body.dateFin).add(1, 'days'));
        const clotures = await Cloture.find({
            date: {
                $gte: dateDebut,
                $lt: dateFin
            }
        });
        res.status(200).json({
            data: clotures,
            status: 'success'
        });
    } catch (err) {
        res.status(404).json({
            status: 'fail',
            message: err
        });
        console.log(err)
    }
};

/**

 * @param {object} req - non utilisé
 * @param {object} res - contient les données demandées

const getEditionsDatas = async (req, res) => {
    try {
        const titres = await Titre.find();
        res.status(200).json({
            data: titres,
            status: 'success'
        });
    } catch (err) {
